Due tipi di socket: TCP e UDP

Fasi:

![Untitled](socket1.png)

- socket() - creazione della socket
    
    ```c
    int s_listen = socket(family, type, protocol);
    // restituisce l'handler della socket
    // family: AF_INET (IPv4)
    // type: SOCK_STREAM o SOCKDGRAM (TCP o UDP)
    // protocol: 0
    ```
    
- bind() - associa i parametri alla socket
    
    ```c
    bind(s_listen, localAdd, addLength)
    //addLenght: lunghezza della variabile localAdd
    //localAdd
    struct localAdd{
    	u_char sin_len;           //lunghezza degli indirizzi
    	u_char sin_family;        //famiglia degli indirizzi
    	u_short sin_port;         //numero di porta
    	struct in_addr sin_addr;  //indirizzo IP
    	char sin_zero[8]          //ignorare
    }
    
    struct localAdd sockAdd;
    sockAdd.sin_family = AF_INET;
    inet_pton(AF_INET, "127.0.0.1", &sockAdd.sin_addr.s_addr) //imposta l'indirizzo IP
    sockAdd.sin_port = htons(9999)       //imposta la porta
    ```
    
- listen() - settare una coda in cui metter in attesa le richieste di connessione
    
    ```c
    int status = listen(s_listen, queueLength);
    // status: -1 errore, 0 altrimenti
    // queueLength: numero di client che possono stare in attesa
    // non bloccante
    ```
    
- accept() -
    
    ```c
    int s_new = accept(s_listen, &clientAddress, &addLength);
    // s_new: nuova socket per la comunicazione
    // clientAddress: struct localAdd, indirizzo del client
    // addLength: dimensione della variabile clientAddress
    // bloccante
    ```
