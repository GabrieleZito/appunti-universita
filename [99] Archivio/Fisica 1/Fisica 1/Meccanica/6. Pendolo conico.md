$$
r=L\sin\theta
$$

$$
\tan\theta=\frac{T\sin\theta}{T\cos\theta}=\frac {F_C}P=\frac{m\omega^2r}{mg}=\frac{\omega^2r}{g}=\frac{\omega^2L\sin\theta}{g}
$$

$$
\frac{\omega^2L\sin\theta}{g}=\frac{\sin\theta}{\cos\theta} \qquad\Rightarrow\qquad\cos\theta=\frac{g}{\omega^2L}\qquad \theta=\arccos\bigg(\frac{g}{\omega^2L}\bigg)
$$

$$
\omega^{-2}=\frac{L\cos\theta}{g}\quad \Rightarrow \quad\bigg(\frac{T}{2\pi}\bigg)^2=\frac{L\cos\theta}{g}\quad\Rightarrow\quad T=2\pi\sqrt{\frac{L\cos\theta}{g}}
$$

$$
f_t=\text{tensione del filo}=P+F_C
$$

$$
|f_t|=\sqrt{P^2+F_C^2}=\sqrt{(mg)^2+(m\omega^2r)^2}
$$
