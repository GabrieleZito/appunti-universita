
$$
\frac{d}{dx}x^n=nx^{n-1},\ n\in\mathbb{N}
$$

$$
\frac{d}{dx}\lg_ax=\frac{1}{x}\lg_ae,\quad \forall x>0
$$

$$
\frac{d}{dx}e^x=e^x
$$

$$
\frac{d}{dx}a^x=a^x\lg a,\quad a>1
$$

$$
\frac{d}{dx}\sin x=\cos x\qquad \frac{d}{dx}\cos x=-\sin x
$$

$$
\frac{d}{dx}\tan x=\frac{1}{\cos^2x}\qquad \frac{d}{dx}\arcsin x=\frac{1}{\sqrt{1-x^2}}\qquad\frac{d}{dx}\arccos x=-\frac{1}{\sqrt{1-x^2}}
$$

$$
\frac{d}{dx}\arctan x=\frac{1}{1+x^2}
$$

$$
\frac{d}{dx}\sinh x=\cosh x\qquad\frac{d}{dx}\cosh x=-\sinh x\qquad\frac{d}{dx}\tanh x=\frac{1}{\cosh x} 
$$
