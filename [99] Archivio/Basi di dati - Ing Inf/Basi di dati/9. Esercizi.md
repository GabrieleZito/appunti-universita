CREATE VIEW Esami_Previsti AS
	SELECT C.Cod_Corso, C.Nome_Corso, SUM(Id_Studente) as Numero_Studenti_Previsti
	FROM PianoStudio P, Corso C
	WHERE equijoin

CREATE VIEW Esami_Superati AS
	SELECT C.Cod_Corso, C.Nome_Corso, SUM(Id_Studente) as Numero_Esami_Superati
	FROM Esame E, Corso C
	WHERE equijoin
